# Installation
  
Dependencies:

  sudo apt-get install build-essential freeglut3-dev libfreeimage-dev libgl1-mesa-dev \
                       libopenal-dev libpango1.0-dev libsdl-ttf2.0-dev libsndfile-dev \
                       libxinerama-dev

Gem:
  gem install gosu


# Usage
$ git clone https://github.com/eladmeidar/picho_game.git
$ cd picho_game
$ ruby picho_game

Enjoy
